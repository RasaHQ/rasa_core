    the `name` method."""

        self._name = name


[docs]    def run(self, dispatcher, tracker, domain):
        """Simple run implementation uttering a (hopefully defined) template."""

        dispatcher.utter_template(self.name(),
                                  filled_slots=tracker.current_slot_values())
        return []
